\newcommand{\nHeader}[3]
{
	\title{\textbf{#1}}
	\author{#2}
	\lhead{#3}
	\cfoot{\thepage\ of \pageref{LastPage}}
}

\newcommand{\nTitle}[1]
{
	\newpage			
	\vspace*{\fill}		
	\begin{center}	
		\part{#1}		
	\end{center}
	\vspace*{\fill}		
	\newpage			
	\setcounter{section}{1}
}

\newenvironment{nAbstract}		
{								
	\vspace*{\fill}				
	\begin{center}				
		\begin{abstract}		
}
{
		\end{abstract}			
	\end{center}				
	\vspace*{\fill}				
}

\newcommand{\nImage}[3] 
{
\begin{figure}[H]
\centering
	\includegraphics[keepaspectratio=true, width=#3\textwidth]{#1}
	\caption{#2}
\end{figure}
}

\newcommand{\nCode}[2]
{
	\paragraph{#1}~\\
	\lstinputlisting{#2}
}
